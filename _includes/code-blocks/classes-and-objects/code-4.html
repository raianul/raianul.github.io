class Driver:
    def __init__(self, name, phone, plate, rating=5.0):
        self.name = name
        self.phone = phone
        self.plate = plate
        self.rating = rating
        self.active = False

    def start_trip(self, destination):
        self.active = True
        print(f"{self.name} is now heading to {destination}.")

    def end_trip(self):
        self.active = False
        print(f"{self.name}'s trip has ended.")

    def update_rating(self, new_rating):
        self.rating = (self.rating + new_rating) / 2
        print(f"{self.name}'s new rating: {self.rating:.1f}")
