class Pot:
    def __init__(self, color, height, price):
        self.color = color
        self.height = height
        self.price = price

    def describe(self):
        print(f"Color: {self.color}, Height: {self.height}cm, Price: {self.price}")

    def sell(self):
        print(f"The {self.color} pot has been sold!")
